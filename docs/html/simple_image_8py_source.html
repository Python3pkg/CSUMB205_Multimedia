292  def cdist(self, **kwargs):
293  """
294  Gets the color distance between this pixel and a specified color or pixel object
295  ARGS:
296  color tuple, 3 integer values where 0<=x<=255
297  pixel an rgbPixel object
298  RETURNS:
299  A Float, that is this pixel's color's distance to the given rgbPixel or
300  color tuple.
301  A negative number means that this pixel will likely be lighter, while
302  a positive number means that this pixel will likely be darker
303  """
304  if not any( [elem in kwargs for elem in ["color", "pixel"]] ):
305  raise KeyError("You must pass either a color or pixel through the keywords 'color' or 'pixel'")
306  if "color" in kwargs:
307  self.__isColor(kwargs['color'])
308  elif "pixel" in kwargs:
309  self.__isPixel(kwargs['pixel'])
310  #END IF/ELIF
311  if "color" in kwargs:
312  givenR, givenG, givenB = kwargs['color']
313  elif "pixel" in kwargs:
314  givenR, givenG, givenB = kwargs['pixel']._color_as_tuple()
315  rDist = givenR-self.__red

317  bDist = givenB-self.__blue
